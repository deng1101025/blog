# 快排算法

给定数组，快速从小到大排序

var arr1 = [2,4,6,3,6,8,7,5]
function quickSort(arr) {
	if(arr.length <= 1) {
		return arr
	}
	var index = Math.floor(arr.length / 2)
	var pivod = arr.splice(index, 1)[0]
	var prev = []
	var last = []
	arr.forEach(item => {
		if(item > pivod) {
			last.push(item)
		} else {
			prev.push(item)
        }
    })

	return [...quickSort(prev), pivod, ...quickSort(last)]
}
